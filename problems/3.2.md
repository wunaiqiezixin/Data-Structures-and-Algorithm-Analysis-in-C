/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
/* 没有哨兵节点 */
void PrintLots(List L, List P)
{
    Position pP = P;
    Position pL = L;
    int curL = 1;//但前pL指向L的curL个节点
    while (pP && pL)
    {
        int target = pP->val;
        while (target <= 0 && pP)
        {
            pP = pP->next;
            if (pP) //防止访问空指针
                target = pP->val;
        }
        if (!pP)
            break;
        while (pL && curL < target)
        {
            pL = pL->next;
            curL++;
        }
        if (pL)
            cout << pL->val << " ";
        pP = pP->next;
    }
}
