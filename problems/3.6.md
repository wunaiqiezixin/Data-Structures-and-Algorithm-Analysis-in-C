/**
 * Definition for singly-linked list.
 * typedef Node* PtrToNode;
 * typedef PtrToNode List;
 * typedef PtrToNode Poly;
 * struct Node
 * {
 *    int Coefficient;
 *    int Exponent;
 *    PtrToNode Next;
 *    Node(): Coefficient(-1), Exponent(-1), Next(nullptr) {}
 *    Node(PtrToNode ptr): Coefficient(ptr->Coefficent), 
 *                          Exponent(ptr->Exponent), Next(nullptr) {}
 *    Node(int coeff, int expo): Coefficient(coeff), Exponent(expo), 
 *                                  Next(nullptr) {}
 * };
 */
Poly Add(Poly P1, Poly P2)
{
    //哨兵节点
    Poly newPoly = new Node();
    PtrToNode cur = newPoly;
    PtrToNode ptr1 = P1, ptr2 = P2;
    while (ptr1 && ptr2)
    {
        if (ptr1->Exponent > ptr2->Exponent)
        {
            if (ptr1->Coefficient != 0)
            {
                cur->Next = new Node(ptr1);
                cur = cur->Next;
            }
            ptr1 = ptr1->Next;
        }
        else if (ptr1->Exponent < ptr2->Exponent)
        {
            if (ptr2->Coefficient != 0)
            {
                cur->Next = new Node(ptr2);
                cur = cur->Next;
            }
            ptr2 = ptr2->Next;
        }
        else
        {
            int coefficient = ptr1->Coefficient + ptr2->Coefficient;
            if (coefficient != 0) //系数和为零，抵消掉，不创建节点
            {
                cur->Next = new Node(coefficient, ptr1->Exponent);
                cur = cur->Next;
            }
            ptr1 = ptr1->Next;
            ptr2 = ptr2->Next;
        }
    }
    while (ptr1)
    {
        if (ptr1->Coefficient != 0)//避免出现系数为0的情况
        {
            cur->Next = new Node(ptr1);
            cur = cur->Next;
        }
        ptr1 = ptr1->Next;
    }
    while (ptr2)
    {
        if (ptr2->Coefficient != 0)
        {
            cur->Next = new Node(ptr2);
            cur = cur->Next;
        }
        ptr2 = ptr2->Next;
    }
    return newPoly->Next;
}
