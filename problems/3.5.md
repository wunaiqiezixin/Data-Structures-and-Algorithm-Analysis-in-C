/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 * typedef ListNode* ptrToNode;
 * typedef ptrToNode List;
 * typedef ptrToNode Position;
 */
List Union(List L1, List L2)//按升序排序
{
    //哨兵节点
    List dummy = new ListNode();
    Position cur = dummy;
    Position pL1 = L1, pL2 = L2;
    while (pL1 && pL2)
    {
        if (pL1->val < pL2->val)
        {
            cur->next = new ListNode(pL1->val);
            cur = cur->next;
            pL1 = pL1->next;
        }
        else if (pL1->val > pL2->val)
        {
            cur->next = new ListNode(pL2->val);
            cur = cur->next;
            pL2 = pL2->next;
        }
        else
        {
            cur->next = new ListNode(pL1->val);
            cur = cur->next;
            pL1 = pL1->next;
            pL2 = pL2->next;
        }
    }
    while (pL1)
    {
        cur->next = new ListNode(pL1->val);
        cur = cur->next;
        pL1 = pL1->next;
    }
    while (pL2)
    {
        cur->next = new ListNode(pL2->val);
        cur = cur->next;
        pL2 = pL2->next;
    }
    return dummy->next;
}
