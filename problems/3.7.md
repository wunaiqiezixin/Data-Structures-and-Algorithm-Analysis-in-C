/**
 * Definition for singly-linked list.
 * typedef Node* PtrToNode;
 * typedef PtrToNode List;
 * typedef PtrToNode Poly;
 * struct Node
 * {
 *    int Coefficient;
 *    int Exponent;
 *    PtrToNode Next;
 *    Node(): Coefficient(-1), Exponent(-1), Next(nullptr) {}
 *    Node(PtrToNode ptr): Coefficient(ptr->Coefficent), 
 *                          Exponent(ptr->Exponent), Next(nullptr) {}
 *    Node(int coeff, int expo): Coefficient(coeff), Exponent(expo), 
 *                                  Next(nullptr) {}
 * };
 */
Poly Multiply(Poly P1, Poly P2)
{
    Poly newPoly = new Node();
    PtrToNode cur = newPoly;

    PtrToNode ptr1, ptr2;

    for (ptr1 = P1; ptr1; ptr1 = ptr1->Next)
    {
        for (ptr2 = P2; ptr2; ptr2 = ptr2->Next)
        {
            int coeff = ptr1->Coefficient * ptr2->Coefficient;
            int expo = ptr1->Exponent + ptr2->Exponent;
            if (coeff != 0)
            {
                //从头寻找插入位置
                cur = newPoly;
                while (cur->Next && 
                    cur->Next->Exponent > expo)
                    cur = cur->Next;
                if (cur->Next && expo == cur->Next->Exponent)
                {
                    cur->Next->Coefficient += coeff;
                    if (cur->Next->Coefficient == 0)
                    {
                        PtrToNode tmp = cur->Next;
                        cur->Next = cur->Next->Next;
                        delete tmp;
                    }
                }
                else
                {
                    PtrToNode newNode = new Node(coeff, expo);
                    newNode->Next = cur->Next;
                    cur->Next = newNode;
                }
            }
        }
    }
    return newPoly->Next;
}
